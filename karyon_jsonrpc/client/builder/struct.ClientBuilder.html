Expand description

Builder for constructing an RPC Client.

§Example

use karyon_jsonrpc::client::ClientBuilder;

async {
    let client = ClientBuilder::new("tcp://127.0.0.1:60000")
        .expect("create builder")
        .set_timeout(5000)
        .build().await
        .expect("build client");

    let result: i32 = client
        .call("Calc.add", (1, 2))
        .await
        .expect("call method");
};
